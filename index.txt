<script>
const S = { view:'loading', activites:[], effectif:[], user:null, statuts:{}, totalJours:0 };const $ = id => document.getElementById(id);

function render() {
  const app = $('app');
  if      (S.view === 'login') renderLogin(app);
  else if (S.view === 'main')  renderMain(app);
  else app.innerHTML = `<div class="loading"><div class="spinner"></div><p>Chargement…</p></div>`;
}

function toast(msg, type='') {
  const t = $('toast');
  t.textContent = msg; t.className = 'toast '+type+' show';
  setTimeout(() => t.className='toast', 3200);
}

function initials(n) { return n.trim().split(/\s+/).map(w=>w[0]).join('').slice(0,2).toUpperCase(); }

function parseDeadline(str) {
  if (!str) return null;
  const m = str.match(/^(\d{1,2})\/(\d{1,2})\/(\d{4})\s+(\d{1,2}):(\d{2})/);
  if (m) return new Date(+m[3], +m[2]-1, +m[1], +m[4], +m[5]).getTime();
  return null;
}

function isDeadlinePassed(a) {
  if (a.deadlineTs) return Date.now() > a.deadlineTs;
  const ts = parseDeadline(a.deadline);
  if (ts) return Date.now() > ts;
  return false;
}

// ── LOGIN ──────────────────────────────────────────────────
function renderLogin(app) {
  app.innerHTML = `
    <div class="login-wrap"><div class="login-card">
      <div class="l-ico">🎖️</div>
      <h2>Identification</h2>
      <p class="l-sub">Saisissez votre adresse mail pour accéder à vos activités</p>
      <input type="email" id="inp-email" placeholder="votre.mail@exemple.com"
        oninput="onEmailInput()" onkeydown="if(event.key==='Enter')login()" autocomplete="email"/>
      <div class="input-error" id="email-err"></div>
      <button class="btn btn-primary" id="btn-login" onclick="login()" disabled>Accéder →</button>
    </div></div>`;
}

function onEmailInput() {
  $('btn-login').disabled = !$('inp-email').value.includes('@');
  $('email-err').textContent = '';
}

function login() {
  const email = $('inp-email').value.trim().toLowerCase();
  if (!email) return;
  const user = S.effectif.find(p => p.email === email);
  if (!user) { $('email-err').textContent = '❌ Adresse mail non reconnue — contactez votre responsable.'; return; }
  S.user = user;
  $('app').innerHTML = `<div class="loading"><div class="spinner"></div><p>Chargement…</p></div>`;
  google.script.run
    .withSuccessHandler(res => {
      if (res.success) { S.statuts = res.statuts; S.totalJours = res.totalJours; S.view = 'main'; render(); }
      else { toast('Erreur : '+res.error,'err'); S.view='login'; render(); }
    })
    .withFailureHandler(() => { toast('Erreur de connexion','err'); S.view='login'; render(); })
    .getStatutsByEmail(email);
}

// ── MAIN ──────────────────────────────────────────────────
function renderMain(app) {
  const nb = S.activites.length;
  app.innerHTML = `
    <div class="main">
      <div class="user-bar">
        <div class="u-info">
          <div class="u-avatar">${initials(S.user.nom)}</div>
          <div>
            <div class="u-name">${esc(S.user.nom)}</div>
            <div class="u-grade">${esc(S.user.grade||'Personnel')}</div>
          </div>
        </div>
        <div style="display:flex; align-items:center; gap:8px;">
          <div class="u-counter" id="web-total-jours">⏱ ${S.totalJours > 0 ? S.totalJours + ' jours' : '0 jour'}</div>
          <button class="btn-change" onclick="logout()">Changer</button>
        </div>
      </div>
      <div class="sec-title">${nb} activité${nb!==1?'s':''}</div>
      ${nb===0
        ? `<div class="empty"><div class="e-ico">📋</div><h3>Aucune activité</h3><p>L'administrateur n'a pas encore ajouté d'activités.</p></div>`
        : `<div class="grid">${S.activites.map(cardHtml).join('')}</div>`}
    </div>`;
}

function cardHtml(a) {
  const statut = S.statuts[a.id] || 'En attente';
  const locked = isDeadlinePassed(a);
  const isMission = (a.type === 'Mission');

  let cardCls = '';
  if (locked) {
    cardCls = 's-locked';
  } else if (!isMission) {
    cardCls = {'Présent':'s-present','Absent':'s-absent','En mission':'s-mission'}[statut]||'';
  } else if (statut !== 'En attente' && statut !== 'Indisponible') {
    cardCls = 's-selected';
  }

  const badgeHtml = isMission 
    ? `<span class="acard-badge mission-badge">Mission</span>` 
    : `<span class="acard-badge">Activités</span>`;

  let datesHtml = '';
  if (a.debut || a.fin) {
    datesHtml = `<div class="acard-dates">`;
    if (a.debut) datesHtml += `<div class="date-item"><span class="date-label">Début</span><span class="date-val">▶ ${esc(a.debut)}</span></div>`;
    if (a.fin)  datesHtml += `<div class="date-item"><span class="date-label">Fin</span><span class="date-val">⏹ ${esc(a.fin)}</span></div>`;
    if (isMission && a.nbSemaines) datesHtml += `<span class="date-duree" style="background:#e3f2fd; color:#0d47a1;">⏱ ${esc(a.nbSemaines)} sem.</span>`;
    else if (a.duree) datesHtml += `<span class="date-duree">⏱ ${esc(a.duree)} j</span>`;
    datesHtml += `</div>`;
  }

  const descHtml = a.description ? `<div class="acard-desc">${esc(a.description)}</div>` : '';

  let deadlineHtml = '';
  if (a.deadline) {
    const cls = locked ? 'passed' : 'ok';
    const ico = locked ? '🔒' : '⚠️';
    deadlineHtml = `<div class="acard-deadline ${cls}">${ico} Deadline : ${esc(a.deadline)}</div>`;
  }

  let statutZone = '';
  if (locked) {
    const label = statut !== 'En attente'
      ? `Statut enregistré : <strong>${esc(statut)}</strong>`
      : 'Aucun statut enregistré';
    statutZone = `<div class="locked-banner">🔒 Positionnement clôturé<br><small style="font-weight:400">${label}</small></div>`;
  } else {
    if (isMission) {
      const totalWeeks = parseInt(a.nbSemaines) || 1;
      let btns = '';
      
      const currentWeeks = statut.split(',').map(x => x.trim().toUpperCase());
      const isIndispo = statut === 'Indisponible';

      for (let w = 1; w <= totalWeeks; w++) {
        const wKey = 'S' + w;
        const active = !isIndispo && currentWeeks.includes(wKey);
        btns += `<button class="sbtn week${active?' active':''}" onclick="toggleMissionWeek('${esc(a.id)}','${wKey}')">${wKey}</button>`;
      }
      
      btns += `<button class="sbtn indispo${isIndispo?' active':''}" onclick="setMissionIndisponible('${esc(a.id)}')">🚫 Indisponible</button>`;
      
      statutZone = `<div class="s-label">Vos disponibilités</div><div class="s-grid mission-grid" id="btns-${esc(a.id)}">${btns}</div>`;
    } else {
      const BTNS = [
        {key:'Présent',    c:'present', ico:'✅'},
        {key:'Absent',     c:'absent',  ico:'❌'},
        {key:'En mission', c:'mission', ico:'🚁'},
        {key:'En attente', c:'attente', ico:'⏳'}
      ];
      const btns = BTNS.map(b => {
        const active = statut === b.key;
        return `<button class="sbtn ${b.c}${active?' active':''}"
          onclick="setStatut('${esc(a.id)}','${escQ(b.key)}')"
          ${active?'disabled':''}>${b.ico} ${b.key}</button>`;
      }).join('');
      statutZone = `<div class="s-label">Votre statut</div><div class="s-grid" id="btns-${esc(a.id)}">${btns}</div>`;
    }
  }

  return `
    <div class="acard ${cardCls}" id="card-${esc(a.id)}">
      ${badgeHtml}
      <div class="acard-name">${esc(a.nom)}</div>
      ${datesHtml}
      ${descHtml}
      ${deadlineHtml}
      <hr class="acard-sep">
      ${statutZone}
    </div>`;
}

function setStatut(activiteId, statut) {
  const a = S.activites.find(x => x.id === activiteId);
  if (a && isDeadlinePassed(a)) { toast('🔒 Délai dépassé — positionnement clôturé','warn'); return; }

  const btnsEl = $('btns-'+activiteId);
  if (btnsEl) btnsEl.querySelectorAll('button').forEach(b => b.disabled=true);

  google.script.run
    .withSuccessHandler(res => {
      if (res.success) {
        S.statuts[activiteId] = statut;
        
        // ─── ICI : LES 3 LIGNES AJOUTÉES CONTRE LE COMPTEUR ───
        S.totalJours = res.totalJours;
        const counterEl = $('web-total-jours');
        if (counterEl) counterEl.textContent = '⏱ ' + (S.totalJours > 0 ? S.totalJours + ' jours' : '0 jour');
        // ─────────────────────────────────────────────────────

        const cardEl = $('card-'+activiteId);
        if (cardEl && a) cardEl.outerHTML = cardHtml(a);
        toast('✅ Statut mis à jour','ok');
      } else if (res.error === 'deadline') {
        toast('🔒 Délai dépassé — positionnement clôturé','warn');
        if (btnsEl) btnsEl.querySelectorAll('button').forEach(b => b.disabled=false);
      } else {
        toast('❌ Erreur : '+res.error,'err');
        if (btnsEl) btnsEl.querySelectorAll('button').forEach(b => b.disabled=false);
      }
    })
    .withFailureHandler(() => {
      toast('❌ Erreur de connexion','err');
      if (btnsEl) btnsEl.querySelectorAll('button').forEach(b => b.disabled=false);
    })
    .setStatut(S.user.email, activiteId, statut);
}

function toggleMissionWeek(activiteId, weekKey) {
  let currentStatut = S.statuts[activiteId] || 'En attente';
  let arr = currentStatut.split(',').map(x => x.trim()).filter(x => x && x !== 'En attente' && x !== 'Indisponible');
  
  const idx = arr.indexOf(weekKey);
  if (idx > -1) {
    arr.splice(idx, 1);
  } else {
    arr.push(weekKey);
  }
  
  arr.sort((a,b) => parseInt(a.replace('S','')) - parseInt(b.replace('S','')));
  
  const nouveauStatut = arr.length > 0 ? arr.join(', ') : 'En attente';
  setStatut(activiteId, nouveauStatut);
}

function setMissionIndisponible(activiteId) {
  setStatut(activiteId, 'Indisponible');
}

function logout() { S.user=null; S.statuts={}; S.view='login'; render(); }

function esc(s) {
  return String(s).replace(/&/g,'&amp;').replace(/</g,'&lt;').replace(/>/g,'&gt;')
    .replace(/"/g,'&quot;').replace(/'/g,'&#39;');
}
function escQ(s) { return String(s).replace(/'/g,"\\'"); }

  })
  .getInitData();
