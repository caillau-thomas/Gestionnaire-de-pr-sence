<script>
// ─── État ────────────────────────────────────────────────────
const S = {
  view      : 'loading',  // loading | login | main
  activites : [],
  effectif  : [],
  user      : null,
  statuts   : {}          // activiteId -> statut
};

// ─── Utilitaires ─────────────────────────────────────────────
const $ = id => document.getElementById(id);

function render() {
  const app = $('app');
  if      (S.view === 'login') renderLogin(app);
  else if (S.view === 'main')  renderMain(app);
  else app.innerHTML = `<div class="loading"><div class="spinner"></div><p>Chargement…</p></div>`;
}

function toast(msg, type = '') {
  const t = $('toast');
  t.textContent = msg;
  t.className   = 'toast ' + type + ' show';
  setTimeout(() => { t.className = 'toast'; }, 2600);
}

function initials(nom) {
  return nom.trim().split(/\s+/).map(w => w[0]).join('').slice(0, 2).toUpperCase();
}

// ─── Vue login ───────────────────────────────────────────────
function renderLogin(app) {
  app.innerHTML = `
    <div class="login-wrap">
      <div class="login-card">
        <div class="l-ico">🎖️</div>
        <h2>Identification</h2>
        <p class="l-sub">Saisissez votre adresse mail pour accéder à vos activités</p>
        <input type="email" id="inp-email" placeholder="votre.mail@exemple.com"
          oninput="onEmailInput()" onkeydown="if(event.key==='Enter')login()" autocomplete="email" />
        <div class="input-error" id="email-err"></div>
        <button class="btn btn-primary" id="btn-login" onclick="login()" disabled>
          Accéder →
        </button>
      </div>
    </div>`;
}

function onEmailInput() {
  const val = $('inp-email').value.trim();
  $('btn-login').disabled = !val.includes('@');
  $('email-err').textContent = '';
}

function login() {
  const email = $('inp-email').value.trim().toLowerCase();
  if (!email) return;
  const user = S.effectif.find(p => p.email === email);
  if (!user) {
    $('email-err').textContent = '❌ Adresse mail non reconnue — contactez votre responsable.';
    return;
  }

  S.user = user;
  $('app').innerHTML = `<div class="loading"><div class="spinner"></div><p>Chargement des statuts…</p></div>`;

  google.script.run
    .withSuccessHandler(res => {
      if (res.success) { S.statuts = res.statuts; S.view = 'main'; render(); }
      else             { toast('Erreur : ' + res.error, 'err'); S.view = 'login'; render(); }
    })
    .withFailureHandler(() => { toast('Erreur de connexion', 'err'); S.view = 'login'; render(); })
    .getStatutsByEmail(email);
}

// ─── Vue principale ──────────────────────────────────────────
function renderMain(app) {
  const u  = S.user;
  const nb = S.activites.length;

  let content = '';
  if (nb === 0) {
    content = `
      <div class="empty">
        <div class="e-ico">📋</div>
        <h3>Aucune activité</h3>
        <p>L'administrateur n'a pas encore ajouté d'activités.</p>
      </div>`;
  } else {
    content = `<div class="grid">${S.activites.map(a => cardHtml(a)).join('')}</div>`;
  }

  app.innerHTML = `
    <div class="main">
      <div class="user-bar">
        <div class="u-info">
          <div class="u-avatar">${initials(u.nom)}</div>
          <div>
            <div class="u-name">${esc(u.nom)}</div>
            <div class="u-grade">${esc(u.grade || 'Personnel')}</div>
          </div>
        </div>
        <button class="btn-change" onclick="logout()">Changer d'utilisateur</button>
      </div>
      <div class="sec-title">${nb} activité${nb !== 1 ? 's' : ''}</div>
      ${content}
    </div>`;
}

function cardHtml(a) {
  const statut = S.statuts[a.id] || 'En attente';
  const cls = { 'Présent': 's-present', 'Absent': 's-absent', 'En mission': 's-mission' }[statut] || '';

  const BTNS = [
    { key: 'Présent',    c: 'present', ico: '✅' },
    { key: 'Absent',     c: 'absent',  ico: '❌' },
    { key: 'En mission', c: 'mission', ico: '🚁' },
    { key: 'En attente', c: 'attente', ico: '⏳' }
  ];

  const btns = BTNS.map(b => {
    const active = statut === b.key;
    return `<button
      class="sbtn ${b.c}${active ? ' active' : ''}"
      onclick="setStatut('${esc(a.id)}','${escQ(b.key)}',this)"
      ${active ? 'disabled' : ''}
    >${b.ico} ${b.key}</button>`;
  }).join('');

  return `
    <div class="acard ${cls}" id="card-${esc(a.id)}">
      <div class="acard-head">
        <div class="acard-name">${esc(a.nom)}</div>
        ${a.date ? `<div class="acard-date">📅 ${esc(a.date)}</div>` : ''}
      </div>
      ${a.description ? `<div class="acard-desc">${esc(a.description)}</div>` : '<div class="acard-desc"></div>'}
      <div class="s-label">Votre statut</div>
      <div class="s-grid" id="btns-${esc(a.id)}">${btns}</div>
    </div>`;
}

function setStatut(activiteId, statut) {
  const btnsEl = $('btns-' + activiteId);
  btnsEl.querySelectorAll('button').forEach(b => b.disabled = true);

  google.script.run
    .withSuccessHandler(res => {
      if (res.success) {
        S.statuts[activiteId] = statut;
        const cardEl = $('card-' + activiteId);
        const a = S.activites.find(x => x.id === activiteId);
        if (cardEl && a) cardEl.outerHTML = cardHtml(a);
        toast('✅ Statut mis à jour', 'ok');
      } else {
        toast('❌ Erreur : ' + res.error, 'err');
        btnsEl.querySelectorAll('button').forEach(b => b.disabled = false);
      }
    })
    .withFailureHandler(() => {
      toast('❌ Erreur de connexion', 'err');
      btnsEl.querySelectorAll('button').forEach(b => b.disabled = false);
    })
    .setStatut(S.user.email, activiteId, statut);
}

function logout() { S.user = null; S.statuts = {}; S.view = 'login'; render(); }

// ─── Helpers anti-XSS ────────────────────────────────────────
function esc(s) {
  return String(s)
    .replace(/&/g,'&amp;').replace(/</g,'&lt;').replace(/>/g,'&gt;')
    .replace(/"/g,'&quot;').replace(/'/g,'&#39;');
}
function escQ(s) { return String(s).replace(/'/g,"\\'"); }

// ─── Initialisation ──────────────────────────────────────────
google.script.run
  .withSuccessHandler(res => {
    if (!res.success) {
      $('app').innerHTML = `<div class="loading"><div style="font-size:40px">❌</div><p>${esc(res.error || 'Erreur')}</p></div>`;
      return;
    }
    S.activites = res.activites;
    S.effectif  = res.effectif;
    if (!S.effectif.length) {
      $('app').innerHTML = `<div class="loading"><div style="font-size:40px">⚠️</div><p>Aucun personnel enregistré — contactez l'administrateur.</p></div>`;
      return;
    }
    S.view = 'login';
    render();
  })
  .withFailureHandler(() => {
    $('app').innerHTML = `<div class="loading"><div style="font-size:40px">❌</div><p>Impossible de joindre le serveur.</p></div>`;
  })
  .getInitData();
</script>
